# Check all keys
required = {'time', 'f0_hz', 'voiced', 'voiced_prob'}
for i, r in enumerate(data):
    assert set(r.keys()) == required, f'Record {i} has wrong keys: {r.keys()}'
    assert isinstance(r['time'], (int, float)), f'Record {i} time not number'
    assert isinstance(r['voiced'], bool), f'Record {i} voiced not bool'
    assert isinstance(r['voiced_prob'], (int, float)), f'Record {i} voiced_prob not number'
    assert 0.0 <= r['voiced_prob'] <= 1.0, f'Record {i} voiced_prob out of range: {r[\"voiced_prob\"]}'
