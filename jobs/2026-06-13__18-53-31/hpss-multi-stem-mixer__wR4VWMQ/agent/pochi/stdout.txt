import librosa

y, sr = librosa.load('input.wav', sr=None)
orig_len = len(y)

y_harmonic, y_percussive = librosa.effects.hpss(y, margin=(1.0, 5.0))
y_h_shifted = librosa.effects.pitch_shift(y_harmonic, sr=sr, n_steps=7)
y_p_stretched = librosa.effects.time_stretch(y_percussive, rate=0.85)

def align(signal, target_len):
    if len(signal) > target_len:
        return signal[:target_len]
    return np.pad(signal, (0, target_len - len(signal)))

h = align(y_h_shifted, orig_len)
p = align(y_p_stretched, orig_len)

mix = 0.7 * h + 0.5 * p

# Individual spectral centroids

import numpy as np
import librosa

y, sr = librosa.load('input.wav', sr=None)
y_harmonic, y_percussive = librosa.effects.hpss(y, margin=(1.0, 5.0))

y_h_shifted = librosa.effects.pitch_shift(y_harmonic, sr=sr, n_steps=7)

# Compare FFT magnitude spectra of harmonic before/after shift
S_before = np.abs(librosa.stft(y_harmonic))
S_after = np.abs(librosa.stft(y_h_shifted))

# Average magnitude across frames
avg_before = np.mean(S_before, axis=1)
avg_after = np.mean(S_after, axis=1)

freqs = librosa.fft_frequencies(sr=sr)

# Find peak frequencies
peak_before = freqs[np.argmax(avg_before)]
peak_after = freqs[np.argmax(avg_after)]
print(f'Peak freq before shift: {peak_before:.2f} Hz')
print(f'Peak freq after shift: {peak_after:.2f} Hz')
print(f'Ratio: {peak_after/peak_before:.4f} (expected ~1.498)')

# Compute weighted average frequency (like spectral centroid but on avg spectrum)
sc_before_avg = np.sum(freqs * avg_before) / np.sum(avg_before)
sc_after_avg = np.sum(freqs * avg_after) / np.sum(avg_after)
print(f'SC from avg spectrum before: {sc_before_avg:.2f}')
print(f'SC from avg spectrum after: {sc_after_avg:.2f}')
print(f'Ratio: {sc_after_avg/sc_before_avg:.4f}')
