import librosa
import numpy as np

y, sr = librosa.load('input.wav', sr=None)
n_fft = 2048
hop_length = 512
n_mfcc = 20

mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc, n_fft=n_fft, hop_length=hop_length)
ssm = librosa.segment.recurrence_matrix(mfcc, metric='cosine', mode='affinity', full=True)
ssm = ssm.astype(np.float64)

# Build kernel
L = 64
half = L // 2
sign_mat = np.ones((L, L))
sign_mat[:half, half:] = -1.0
sign_mat[half:, :half] = -1.0
sigma = L / 4.0
ax = np.arange(L) - half + 0.5
gx = np.exp(-0.5 * (ax / sigma) ** 2)
gauss_2d = np.outer(gx, gx)
kernel = sign_mat * gauss_2d

# Compute novelty
n_frames = ssm.shape[0]
novelty = np.zeros(n_frames, dtype=np.float64)
for i in range(n_frames):
    r_start = i - half
    r_end = i + half
    c_start = i - half
    c_end = i + half
    kr_start = max(0, -r_start)
    kc_start = max(0, -c_start)
    kr_end = L - max(0, r_end - n_frames)
    kc_end = L - max(0, c_end - n_frames)
    r_start_c = max(0, r_start)
    r_end_c = min(n_frames, r_end)
    c_start_c = max(0, c_start)
    c_end_c = min(n_frames, c_end)
    novelty[i] = np.sum(ssm[r_start_c:r_end_c, c_start_c:c_end_c] * kernel[kr_start:kr_end, kc_start:kc_end])

# Print novelty curve statistics
novelty_norm = (novelty - novelty.min()) / (novelty.max() - novelty.min() + 1e-10)

# Print top 20 peaks
from scipy.signal import find_peaks
peaks, props = find_peaks(novelty_norm, distance=5, prominence=0.01)
sorted_idx = np.argsort(novelty_norm[peaks])[::-1]
print('Top 20 peaks by height:')
for idx in sorted_idx[:20]:
    frame = peaks[idx]
    time = librosa.frames_to_time(frame, sr=sr, hop_length=hop_length)
    print(f'  frame={frame}, time={time:.3f}s, height={novelty_norm[frame]:.4f}')

# Also check with different prominence thresholds
print()
for prom in [0.01, 0.03, 0.05, 0.08, 0.1]:
    peaks2, _ = find_peaks(novelty_norm, distance=int(1.0*sr/hop_length), prominence=prom, height=0.05)
    margin = int(0.5*sr/hop_length)
    peaks2 = peaks2[(peaks2 > margin) & (peaks2 < n_frames - margin)]
    times = [f'{librosa.frames_to_time(p, sr=sr, hop_length=hop_length):.2f}' for p in peaks2]
    print(f'prom={prom}: {times}')
