n_frames = mfcc.shape[1]

ssm = librosa.segment.recurrence_matrix(mfcc, mode='affinity', metric='cosine', full=True)

# Build checkerboard kernel
L = 64
sigma = L / 4.0
kernel = np.ones((L, L))
kernel[:L//2, L//2:] = -1.0
kernel[L//2:, :L//2] = -1.0
half = L // 2
y_idx, x_idx = np.mgrid[0:L, 0:L]
gaussian = np.exp(-0.5 * ((y_idx - half + 0.5)**2 + (x_idx - half + 0.5)**2) / sigma**2)
kernel *= gaussian
kernel /= np.sqrt(np.sum(kernel**2))

# Compute novelty
novelty = np.zeros(n_frames)
for t in range(n_frames):
    r_start = max(t - half, 0)
    r_end = min(t + half, n_frames)
    c_start = max(t - half, 0)
    c_end = min(t + half, n_frames)
    patch = ssm[r_start:r_end, c_start:c_end]
    kr_start = half - (t - r_start)
    kr_end = half + (r_end - t)
    kc_start = half - (t - c_start)
    kc_end = half + (c_end - t)
    k_patch = kernel[kr_start:kr_end, kc_start:kc_end]
    novelty[t] = np.sum(patch * k_patch)
