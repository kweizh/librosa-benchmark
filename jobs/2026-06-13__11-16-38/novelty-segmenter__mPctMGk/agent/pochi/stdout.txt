import numpy as np
import json
import librosa
import scipy.signal

def make_checkerboard(size):
    half = size // 2
    kernel = np.zeros((size, size))
    kernel[:half, :half] = 1
    kernel[half:, half:] = 1
    kernel[:half, half:] = -1
    kernel[half:, :half] = -1
    window = scipy.signal.windows.gaussian(size, std=size/4.0)
    window_2d = np.outer(window, window)
    return kernel * window_2d

y, sr = librosa.load('/workspace/input.wav', sr=None)
hop_length = 512
mfcc = librosa.feature.mfcc(y=y, sr=sr, hop_length=hop_length)
ssm = librosa.segment.recurrence_matrix(mfcc, mode='affinity', metric='cosine', full=True)

kernel_size = 64
N = ssm.shape[0]
novelty = np.zeros(N)
half = kernel_size // 2
kernel = make_checkerboard(kernel_size)

for i in range(N):
    i_start = max(0, i - half)
    i_end = min(N, i + half)
    k_start = half - (i - i_start)
    k_end = half + (i_end - i)
    patch = ssm[i_start:i_end, i_start:i_end]
    k_patch = kernel[k_start:k_end, k_start:k_end]
    novelty[i] = np.sum(patch * k_patch)

print("Novelty max:", np.max(novelty))
print("Novelty min:", np.min(novelty))

novelty /= np.max(novelty)
peaks, props = scipy.signal.find_peaks(novelty, height=0.1, distance=32, prominence=0.05)
print("Peaks:", peaks)
print("Prominences:", props.get('prominences'))
