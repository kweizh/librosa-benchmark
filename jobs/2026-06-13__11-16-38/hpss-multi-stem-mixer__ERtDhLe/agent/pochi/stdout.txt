import librosa
import soundfile as sf
import numpy as np

def main():
    input_path = '/workspace/input.wav'
    output_path = '/workspace/output.wav'
    
    # 1. Read input.wav preserving native sample rate
    y, sr = librosa.load(input_path, sr=None)
    
    # 2. Decompose into harmonic and percussive
    y_harmonic, y_percussive = librosa.effects.hpss(y, margin=(1.0, 5.0))
    
    # 3. Pitch-shift harmonic up by 7 semitones
    y_harmonic_shifted = librosa.effects.pitch_shift(y_harmonic, sr=sr, n_steps=7)
    
    # 4. Time-stretch percussive by a rate of 0.85
    y_percussive_stretched = librosa.effects.time_stretch(y_percussive, rate=0.85)
    
    # 5. Re-align lengths to original input length
    y_harmonic_shifted = librosa.util.fix_length(y_harmonic_shifted, size=len(y))
    y_percussive_stretched = librosa.util.fix_length(y_percussive_stretched, size=len(y))
    
    # Mix
    mix = 0.7 * y_harmonic_shifted + 0.5 * y_percussive_stretched
    
    # 6. Trim leading and trailing silence
    y_trimmed, _ = librosa.effects.trim(mix, top_db=30)
    
    # 7. Write result
    sf.write(output_path, y_trimmed, sr)

if __name__ == '__main__':
    main()
