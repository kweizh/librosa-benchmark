import librosa
import numpy as np
import json

# Load audio
y, sr = librosa.load('/workspace/input.wav', sr=None)

# Parameters
hop_length = 512
n_fft = 2048

fmin = librosa.note_to_hz('C1') # ~32.7 Hz
cqt_n_bins = 84
cqt_bins_per_octave = 12

vqt_n_bins = 84
vqt_bins_per_octave = 12

# STFT
D_stft = librosa.stft(y=y, n_fft=n_fft, hop_length=hop_length)
stft_mag = np.abs(D_stft)
stft_db = librosa.amplitude_to_db(S=stft_mag, ref=np.max, top_db=100.0)

# CQT
D_cqt = librosa.cqt(y=y, sr=sr, hop_length=hop_length, fmin=fmin, n_bins=cqt_n_bins, bins_per_octave=cqt_bins_per_octave)
cqt_mag = np.abs(D_cqt)
cqt_db = librosa.amplitude_to_db(S=cqt_mag, ref=np.max, top_db=100.0)

# VQT
D_vqt = librosa.vqt(y=y, sr=sr, hop_length=hop_length, fmin=fmin, n_bins=vqt_n_bins, bins_per_octave=vqt_bins_per_octave)
vqt_mag = np.abs(D_vqt)
vqt_db = librosa.amplitude_to_db(S=vqt_mag, ref=np.max, top_db=100.0)

# Check shapes
n_frames = stft_db.shape[1]
assert cqt_db.shape[1] == n_frames
assert vqt_db.shape[1] == n_frames

# Frequencies
stft_freqs = librosa.fft_frequencies(sr=sr, n_fft=n_fft)
cqt_freqs = librosa.cqt_frequencies(n_bins=cqt_n_bins, fmin=fmin, bins_per_octave=cqt_bins_per_octave)
vqt_freqs = librosa.interval_frequencies(n_bins=vqt_n_bins, fmin=fmin, intervals='equal', bins_per_octave=vqt_bins_per_octave)

# Save npz
np.savez('/workspace/spec_stack.npz', stft_db=stft_db, cqt_db=cqt_db, vqt_db=vqt_db)

# Save metadata
meta = {
    'n_frames': int(n_frames),
    'hop_length': int(hop_length),
    'sample_rate': float(sr),
    'stft_freqs': stft_freqs.tolist(),
    'cqt_freqs': cqt_freqs.tolist(),
    'vqt_freqs': vqt_freqs.tolist(),
    'n_fft': int(n_fft),
    'cqt_n_bins': int(cqt_n_bins),
    'cqt_bins_per_octave': int(cqt_bins_per_octave),
    'vqt_n_bins': int(vqt_n_bins),
    'vqt_bins_per_octave': int(vqt_bins_per_octave)
}

with open('/workspace/spec_meta.json', 'w') as f:
    json.dump(meta, f)

print(f"Dynamic ranges: STFT {stft_db.max() - stft_db.min()}, CQT {cqt_db.max() - cqt_db.min()}, VQT {vqt_db.max() - vqt_db.min()}")
