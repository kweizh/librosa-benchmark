import librosa
import numpy as np
import json

# 1. Read the input WAV file
y, sr = librosa.load('/workspace/input.wav')
hop_length = 512
duration = librosa.get_duration(y=y, sr=sr)

# 2. Get beat track
tempo, beats = librosa.beat.beat_track(y=y, sr=sr, hop_length=hop_length)

# 3. Compute chroma_cqt
chroma = librosa.feature.chroma_cqt(y=y, sr=sr, hop_length=hop_length)

# 4. Sync chroma to beats
chroma_sync = librosa.util.sync(chroma, beats, aggregate=np.median)

# 5. Coarse segmentation
k_coarse = 6
coarse_beat_boundaries = librosa.segment.agglomerative(chroma_sync, k=k_coarse)

# coarse_beat_boundaries are indices into the beats array
coarse_frame_boundaries = beats[coarse_beat_boundaries]

# Append the last frame if not present
total_frames = chroma.shape[1]
coarse_frame_boundaries = np.append(coarse_frame_boundaries, total_frames - 1)

# Ensure it starts at 0
if coarse_frame_boundaries[0] != 0:
    coarse_frame_boundaries[0] = 0

# Convert coarse boundaries to time
coarse_times = librosa.frames_to_time(coarse_frame_boundaries, sr=sr, hop_length=hop_length)

# Clamp the last coarse boundary to the audio duration
coarse_times[-1] = duration

# Build coarse output
coarse_out = []
for i in range(len(coarse_times) - 1):
    coarse_out.append({
        "index": i,
        "start": float(coarse_times[i]),
        "end": float(coarse_times[i+1])
    })

# 6. Fine segmentation
# For each coarse segment, use librosa.segment.subsegment with n_segments=3
# "pass the original chroma matrix along with the segment boundary frames"
fine_out = []
fine_idx = 0

for i in range(len(coarse_frame_boundaries) - 1):
    start_frame = coarse_frame_boundaries[i]
    end_frame = coarse_frame_boundaries[i+1]
    
    # Extract exactly 3 fine sub-segments per coarse segment
    # We can pass just this interval to subsegment
    interval_frames = np.array([start_frame, end_frame])
    sub_bounds = librosa.segment.subsegment(chroma, interval_frames, n_segments=3)
    
    sub_times = librosa.frames_to_time(sub_bounds, sr=sr, hop_length=hop_length)
    # Clamp the last fine boundary to the end of the coarse segment
    sub_times[-1] = coarse_out[i]["end"]
    # Clamp the first fine boundary to the start of the coarse segment
    sub_times[0] = coarse_out[i]["start"]
    
    # If subsegment returns fewer than 4 boundaries (i.e. < 3 segments), we need to interpolate
    # to ensure exactly 3 segments
    if len(sub_times) < 4:
        sub_times = np.linspace(sub_times[0], sub_times[-1], 4)
    elif len(sub_times) > 4:
        # Just in case, take exactly 4 evenly spaced if it's messed up, or just the first 4?
        # Actually librosa should return exactly n_segments + 1 boundaries if the interval is large enough.
        sub_times = sub_times[:4]
        sub_times[-1] = coarse_out[i]["end"]
        
    for j in range(3):
        fine_out.append({
            "index": fine_idx,
            "start": float(sub_times[j]),
            "end": float(sub_times[j+1]),
            "parent_index": i
        })
        fine_idx += 1

output = {
    "coarse": coarse_out,
    "fine": fine_out
}

with open('/workspace/hierarchy.json', 'w') as f:
    json.dump(output, f, indent=2)

print("Coarse segments:", len(coarse_out))
print("Fine segments:", len(fine_out))

import librosa
import numpy as np
import json

def main():
    # 1. Read the input WAV file
    y, sr = librosa.load('/workspace/input.wav')
    hop_length = 512
    duration = librosa.get_duration(y=y, sr=sr)

    # 2. Get beat track
    tempo, beats = librosa.beat.beat_track(y=y, sr=sr, hop_length=hop_length)

    # 3. Compute chroma_cqt
    chroma = librosa.feature.chroma_cqt(y=y, sr=sr, hop_length=hop_length)

    # 4. Sync chroma to beats
    chroma_sync = librosa.util.sync(chroma, beats, aggregate=np.median)

    # 5. Coarse segmentation
    k_coarse = 6
    coarse_beat_boundaries = librosa.segment.agglomerative(chroma_sync, k=k_coarse)

    # coarse_beat_boundaries are indices into the beats array
    coarse_frame_boundaries = beats[coarse_beat_boundaries]

    # Append the total number of frames to get the end of the last segment
    total_frames = chroma.shape[1]
    coarse_frame_boundaries = np.append(coarse_frame_boundaries, total_frames)

    # Ensure it starts at 0
    if coarse_frame_boundaries[0] != 0:
        coarse_frame_boundaries[0] = 0

    # 6. Fine segmentation using subsegment
    # Pass the original chroma matrix and the coarse boundaries
    sub_bounds = librosa.segment.subsegment(chroma, coarse_frame_boundaries, n_segments=3)
    
    # sub_bounds contains the left boundaries of all fine segments.
    # We append total_frames to have the right boundary of the last fine segment.
    sub_bounds_full = np.append(sub_bounds, total_frames)

    # Convert to times
    coarse_times = librosa.frames_to_time(coarse_frame_boundaries, sr=sr, hop_length=hop_length)
    # Clamp the last coarse boundary to the audio duration
    coarse_times[-1] = duration
    
    sub_times = librosa.frames_to_time(sub_bounds_full, sr=sr, hop_length=hop_length)
    # Clamp the last fine boundary to the audio duration
    sub_times[-1] = duration

    # Build coarse output
    coarse_out = []
    for i in range(len(coarse_times) - 1):
        coarse_out.append({
            "index": i,
            "start": float(coarse_times[i]),
            "end": float(coarse_times[i+1])
        })

    # Build fine output
    fine_out = []
    fine_idx = 0
    
    # Since subsegment partitions each interval into exactly n_segments (3),
    # we know that fine segments 0,1,2 belong to coarse 0, etc.
    for i in range(len(coarse_out)):
        for j in range(3):
            # The start and end of this fine segment
            start_t = float(sub_times[fine_idx])
            end_t = float(sub_times[fine_idx+1])
            
            # Ensure exact coverage by clamping to coarse boundaries if necessary
            if j == 0:
                start_t = coarse_out[i]["start"]
            if j == 2:
                end_t = coarse_out[i]["end"]
                
            fine_out.append({
                "index": fine_idx,
                "start": start_t,
                "end": end_t,
                "parent_index": i
            })
            fine_idx += 1

    output = {
        "coarse": coarse_out,
        "fine": fine_out
    }

    with open('/workspace/hierarchy.json', 'w') as f:
        json.dump(output, f, indent=2)
