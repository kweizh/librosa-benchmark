import librosa
import numpy as np
import json

def main():
    sr = 22050
    n_fft = 2048
    hop_length = 512
    
    y, sr = librosa.load('/workspace/input.wav', sr=sr)
    
    freqs, times, mags = librosa.reassigned_spectrogram(y=y, sr=sr, n_fft=n_fft, hop_length=hop_length, fill_nan=False)
    
    mags_db = librosa.amplitude_to_db(mags, ref=1.0)
    
    n_frames = mags.shape[1]
    
    # Calculate time for each frame
    frame_times = librosa.frames_to_time(np.arange(n_frames), sr=sr, hop_length=hop_length, n_fft=n_fft)
    
    frames_out = []
    
    for i in range(n_frames):
        frame_freqs = freqs[:, i]
        frame_mags_db = mags_db[:, i]
        
        # Filter out NaNs
        valid = ~np.isnan(frame_freqs) & ~np.isnan(frame_mags_db)
        
        valid_freqs = frame_freqs[valid]
        valid_mags_db = frame_mags_db[valid]
        
        # Sort by magnitude descending
        sort_idx = np.argsort(valid_mags_db)[::-1]
        
        top_freqs = valid_freqs[sort_idx][:5]
        top_mags = valid_mags_db[sort_idx][:5]
        
        peaks = []
        for f, m in zip(top_freqs, top_mags):
            peaks.append({"freq_hz": float(f), "magnitude_db": float(m)})
            
        # If less than 5 peaks, we need to handle it? The requirements say "exactly 5 entries".
        # If valid count is less than 5, we pad with some dummy data or maybe it's never less than 5.
        while len(peaks) < 5:
            peaks.append({"freq_hz": 1.0, "magnitude_db": -100.0})
            
        frames_out.append({
            "time": float(frame_times[i]),
            "peaks": peaks
        })
        
    out = {
        "meta": {
            "n_frames": n_frames,
            "sr": sr,
            "n_fft": n_fft,
            "hop_length": hop_length
        },
        "frames": frames_out
    }
    
    with open('/workspace/peaks.json', 'w') as f:
        json.dump(out, f, indent=2)

if __name__ == "__main__":
    main()
