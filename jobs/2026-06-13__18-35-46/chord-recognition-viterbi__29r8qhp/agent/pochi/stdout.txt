import librosa
import numpy as np
import json

# 1. Load audio
y, sr = librosa.load('/workspace/input.wav')
hop_length = 512

# 2. Compute chroma
chroma = librosa.feature.chroma_cqt(y=y, sr=sr, hop_length=hop_length)

# 3. Define templates
chord_labels = []
notes = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B']
for n in notes:
    chord_labels.append(f"{n}:maj")
for n in notes:
    chord_labels.append(f"{n}:min")

templates = np.zeros((24, 12))
for i in range(12):
    # Major: root, +4, +7
    templates[i, i] = 1
    templates[i, (i + 4) % 12] = 1
    templates[i, (i + 7) % 12] = 1
    
    # Minor: root, +3, +7
    templates[12 + i, i] = 1
    templates[12 + i, (i + 3) % 12] = 1
    templates[12 + i, (i + 7) % 12] = 1

# Normalize templates
templates = templates / np.linalg.norm(templates, axis=1, keepdims=True)

# Normalize chroma
chroma_norm = chroma / (np.linalg.norm(chroma, axis=0, keepdims=True) + 1e-8)

# 4. Observation probabilities
prob = np.dot(templates, chroma_norm)
prob = prob / (np.sum(prob, axis=0, keepdims=True) + 1e-8)

# 5. Transition matrix
p_self = 0.99
p_other = (1.0 - p_self) / 23.0
transition = np.full((24, 24), p_other)
np.fill_diagonal(transition, p_self)

# 6. Viterbi decoding
states = librosa.sequence.viterbi(prob=prob, transition=transition)

# 7. Merge segments
times = librosa.frames_to_time(np.arange(chroma.shape[1] + 1), sr=sr, hop_length=hop_length)
duration = librosa.get_duration(y=y, sr=sr)
times[-1] = duration # clamp to end

segments = []
current_state = states[0]
start_idx = 0

for i in range(1, len(states)):
    if states[i] != current_state:
        segments.append({
            "start": float(times[start_idx]),
            "end": float(times[i]),
            "chord": chord_labels[current_state]
        })
        current_state = states[i]
        start_idx = i

# Last segment
segments.append({
    "start": float(times[start_idx]),
    "end": float(duration),
    "chord": chord_labels[current_state]
})

with open('/workspace/chords.json', 'w') as f:
    json.dump(segments, f, indent=2)

print("Done")
