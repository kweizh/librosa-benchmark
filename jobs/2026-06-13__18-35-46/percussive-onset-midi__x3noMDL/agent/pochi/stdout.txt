import librosa
import numpy as np
import json

def main():
    # 1. Read input
    y, sr = librosa.load('/workspace/input.wav', sr=None)

    # 2. HPSS
    y_harmonic, y_percussive = librosa.effects.hpss(y)

    # 3. Onset detection
    hop_length = 512
    onset_env = librosa.onset.onset_strength(y=y_percussive, sr=sr, hop_length=hop_length)
    onsets_frames = librosa.onset.onset_detect(onset_envelope=onset_env, sr=sr, hop_length=hop_length)

    # 4. Beat tracking
    tempo, beats = librosa.beat.beat_track(onset_envelope=onset_env, sr=sr, hop_length=hop_length)
    if isinstance(tempo, np.ndarray):
        tempo = tempo.item()

    # 16th-note step
    step_16th = 60.0 / tempo / 4.0

    # 5. Snap onsets
    onsets_times = librosa.frames_to_time(onsets_frames, sr=sr, hop_length=hop_length)

    # 6. Velocity
    velocities = onset_env[onsets_frames]
    max_vel = np.max(velocities) if len(velocities) > 0 else 1.0
    if max_vel > 0:
        velocities = velocities / max_vel

    # 7. Write to hits.json
    hits = []
    for t, v in zip(onsets_times, velocities):
        grid_index = int(round(t / step_16th))
        snapped_t = grid_index * step_16th
        hits.append({
            "time_seconds": float(snapped_t),
            "grid_index": int(grid_index),
            "velocity": float(v),
            "raw_time_seconds": float(t)
        })
    
    # Optional: ensure non-decreasing sort by time_seconds and grid_index
    hits.sort(key=lambda x: (x["time_seconds"], x["grid_index"]))

    with open('/workspace/hits.json', 'w') as f:
        json.dump(hits, f, indent=2)

if __name__ == '__main__':
    main()
