import librosa
import numpy as np
import scipy.signal
import scipy.spatial.distance
import json

def make_checkerboard(size):
    half = size // 2
    kernel = np.ones((size, size))
    kernel[:half, half:] = -1
    kernel[half:, :half] = -1
    window = scipy.signal.windows.gaussian(size, std=size/4.0)
    window_2d = np.outer(window, window)
    return kernel * window_2d

def main():

    y, sr = librosa.load('/workspace/input.wav')

    # 2. Compute MFCCs

        if hasattr(ssm, 'toarray'):

    # Wait, `full` is not a standard kwarg in modern librosa. But let's check librosa version. It says v0.11.0.
    
    # Actually let's just make sure it's dense:
    # Let's compute it purely using cdist for the SSM if recurrence_matrix behaves weirdly, but prompt says:
    # "Inspect librosa.feature.mfcc for MFCC extraction and librosa.segment.recurrence_matrix (with mode='affinity', metric='cosine', and full=True) for the dense self-similarity matrix."
    ssm = librosa.segment.recurrence_matrix(mfcc, mode='affinity', metric='cosine', full=True)

    # 4. Construct Foote-style checkerboard kernel
    kernel_size = 64 # e.g. 64 frames
    kernel = make_checkerboard(kernel_size)

    # 5. Convolve kernel along the diagonal of SSM
    n_frames = ssm.shape[0]
    novelty_curve = np.zeros(n_frames)
    half_k = kernel_size // 2

    for i in range(half_k, n_frames - half_k):
        patch = ssm[i - half_k : i + half_k, i - half_k : i + half_k]
        novelty_curve[i] = np.sum(patch * kernel)

    # Normalize novelty curve
    if np.max(novelty_curve) > 0:
        novelty_curve = novelty_curve / np.max(novelty_curve)

    # 6. Pick peaks
    # We can use librosa.util.peak_pick
    # peak_pick(x, pre_max, post_max, pre_avg, post_avg, delta, wait)
    # Or scipy.signal.find_peaks
    peaks, _ = scipy.signal.find_peaks(novelty_curve, distance=kernel_size//2, prominence=0.1)

    # Filter out peaks too close to start/end
    valid_peaks = [p for p in peaks if p > kernel_size and p < n_frames - kernel_size]

    # 7. Convert to timestamps
    timestamps = librosa.frames_to_time(valid_peaks, sr=sr, hop_length=hop_length)

    # 8. Write to JSON
    output = {"boundaries_sec": [float(t) for t in timestamps]}
    with open('/workspace/boundaries.json', 'w') as f:
        json.dump(output, f)

if __name__ == '__main__':
    main()
