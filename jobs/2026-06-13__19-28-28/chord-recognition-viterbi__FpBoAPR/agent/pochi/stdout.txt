with open('/workspace/chords.json') as f:

# Schema validation
pattern = re.compile(r'^[A-G]#?:(maj|min)$')
allowed = {
    'C:maj','C#:maj','D:maj','D#:maj','E:maj','F:maj','F#:maj','G:maj','G#:maj','A:maj','A#:maj','B:maj',
    'C:min','C#:min','D:min','D#:min','E:min','F:min','F#:min','G:min','G#:min','A:min','A#:min','B:min'
}

errors = []
for i, s in enumerate(segments):
    if not isinstance(s['start'], (int, float)):
        errors.append(f'seg {i}: start is not number')
    if not isinstance(s['end'], (int, float)):
        errors.append(f'seg {i}: end is not number')
    if s['start'] >= s['end']:
        errors.append(f'seg {i}: start ({s[\"start\"]}) >= end ({s[\"end\"]})')
    if not pattern.match(s['chord']):
        errors.append(f'seg {i}: chord \"{s[\"chord\"]}\" does not match pattern')
    if s['chord'] not in allowed:
        errors.append(f'seg {i}: chord \"{s[\"chord\"]}\" not in allowed set')

# Check sorted by start
starts = [s['start'] for s in segments]
if starts != sorted(starts):
    errors.append('segments not sorted by start')
