for i, s in enumerate(segs):
    if not isinstance(s['start'], (int, float)):
        errors.append(f'seg[{i}]: start not a number')
    if not isinstance(s['end'], (int, float)):
        errors.append(f'seg[{i}]: end not a number')
    if s['start'] >= s['end']:
        errors.append(f'seg[{i}]: start >= end ({s[\"start\"]} >= {s[\"end\"]})')
    if not isinstance(s['label'], str) or not re.match(r'^[A-Z]$', s['label']):
        errors.append(f'seg[{i}]: label not a single uppercase letter: {s[\"label\"]}')

# Sorted by start

# Coverage: start near 0, end near duration
