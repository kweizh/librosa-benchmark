import numpy as np
import json
import librosa

# ── Load outputs ────────────────────────────────────────────────────────
data = np.load('/workspace/spec_stack.npz')
stft_db = data['stft_db']
cqt_db = data['cqt_db']
vqt_db = data['vqt_db']

with open('/workspace/spec_meta.json') as f:
    meta = json.load(f)

# ── Required keys in NPZ ────────────────────────────────────────────────
assert set(data.keys()) == {'stft_db', 'cqt_db', 'vqt_db'}, f'Keys: {set(data.keys())}'

# ── Required keys in JSON ───────────────────────────────────────────────
required_keys = {
    'n_frames', 'hop_length', 'sample_rate', 'stft_freqs', 'cqt_freqs',
    'vqt_freqs', 'n_fft', 'cqt_n_bins', 'cqt_bins_per_octave',
    'vqt_n_bins', 'vqt_bins_per_octave'
}
assert set(meta.keys()) == required_keys, f'Missing/extra keys: {set(meta.keys()) ^ required_keys}'

# ── Shape constraints ───────────────────────────────────────────────────
n_frames = meta['n_frames']
n_fft = meta['n_fft']
cqt_n_bins = meta['cqt_n_bins']
vqt_n_bins = meta['vqt_n_bins']

assert stft_db.shape == (1 + n_fft // 2, n_frames), f'STFT shape {stft_db.shape}'
assert cqt_db.shape == (cqt_n_bins, n_frames), f'CQT shape {cqt_db.shape}'
assert vqt_db.shape == (vqt_n_bins, n_frames), f'VQT shape {vqt_db.shape}'

# ── Frequency vector lengths ────────────────────────────────────────────
assert len(meta['stft_freqs']) == stft_db.shape[0]
assert len(meta['cqt_freqs']) == cqt_n_bins
assert len(meta['vqt_freqs']) == vqt_n_bins

# ── Frequency vector correctness ────────────────────────────────────────
fmin = float(librosa.note_to_hz('C1'))
expected_cqt = librosa.cqt_frequencies(
    n_bins=cqt_n_bins, fmin=fmin, bins_per_octave=meta['cqt_bins_per_octave']
)
expected_vqt = librosa.cqt_frequencies(
    n_bins=vqt_n_bins, fmin=fmin, bins_per_octave=meta['vqt_bins_per_octave']
)
assert np.allclose(meta['cqt_freqs'], expected_cqt, rtol=1e-10), 'CQT freqs mismatch'
assert np.allclose(meta['vqt_freqs'], expected_vqt, rtol=1e-10), 'VQT freqs mismatch'

# ── Finite everywhere ───────────────────────────────────────────────────
assert np.all(np.isfinite(stft_db)), 'STFT has non-finite values'
assert np.all(np.isfinite(cqt_db)), 'CQT has non-finite values'
assert np.all(np.isfinite(vqt_db)), 'VQT has non-finite values'

# ── Dynamic range > 20 dB ───────────────────────────────────────────────
for name, arr in [('STFT', stft_db), ('CQT', cqt_db), ('VQT', vqt_db)]:
    dr = arr.max() - arr.min()
    assert dr > 20, f'{name} dynamic range {dr:.2f} <= 20'
    print(f'{name}: shape={arr.shape}, dynamic_range={dr:.2f} dB')

# ── JSON numeric types ──────────────────────────────────────────────────
for k, v in meta.items():
    if k.endswith('_freqs'):
        assert isinstance(v, list), f'{k} is not a list'
        assert all(isinstance(x, (int, float)) for x in v), f'{k} has non-numeric'
    else:
        assert isinstance(v, (int, float)), f'{k} is not a plain scalar: {type(v)}'

print()
print('ALL VALIDATION CHECKS PASSED')
