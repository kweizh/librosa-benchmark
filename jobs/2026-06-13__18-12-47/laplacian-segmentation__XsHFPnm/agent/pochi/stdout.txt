R = librosa.segment.recurrence_matrix(beat_chroma, metric='cosine', mode='affinity',
    sym=True, sparse=False, width=rec_width, axis=-1)
R_enh = librosa.segment.path_enhance(R, n=15, window='hann', clip=True)

W = 9
seq = np.zeros((n_beats, n_beats), dtype=np.float32)
for d in range(1, W + 1):
    w = 1.0 - d / (W + 1)
    idx = np.arange(n_beats - d)
    seq[idx, idx+d] = w; seq[idx+d, idx] = w
np.fill_diagonal(seq, 1.0)

A = 0.5 * R_enh + 0.5 * seq
A = 0.5 * (A + A.T)

d_vec = A.sum(axis=1)
d_inv_sqrt = np.where(d_vec > 0, 1.0 / np.sqrt(d_vec), 0.0)
L_sym = np.eye(n_beats) - np.diag(d_inv_sqrt) @ A @ np.diag(d_inv_sqrt)

eigenvalues, eigenvectors = scipy.linalg.eigh(L_sym, subset_by_index=[0, 5])
embedding = eigenvectors[:, 1:6]
embedding = normalize(embedding, norm='l2', axis=1)

# KMeans with 4 clusters – allows label re-use (A/B/A structure)
km = KMeans(n_clusters=4, n_init=20, random_state=42)
km.fit(embedding)

import numpy as np, librosa, librosa.onset, scipy.linalg
from sklearn.cluster import KMeans
from sklearn.preprocessing import normalize

y, sr = librosa.load('/workspace/input.wav', sr=22050, mono=True)
HOP_LENGTH = 512
n_frames = int(len(y) // HOP_LENGTH)
onset_env = librosa.onset.onset_strength(y=y, sr=sr, hop_length=HOP_LENGTH)
tempo, beat_frames = librosa.beat.beat_track(onset_envelope=onset_env, sr=sr,
    hop_length=HOP_LENGTH, start_bpm=120.0, units='frames', sparse=True)
beat_frames = librosa.util.fix_frames(beat_frames, x_min=0, x_max=n_frames-1, pad=True)
n_beats = len(beat_frames)
beat_times = librosa.frames_to_time(beat_frames, sr=sr, hop_length=HOP_LENGTH)

C_cqt = librosa.cqt(y, sr=sr, hop_length=HOP_LENGTH, n_bins=84, bins_per_octave=12)
C_mag = np.abs(C_cqt)
beat_feat = librosa.util.sync(C_mag, beat_frames, aggregate=np.median, pad=True, axis=-1)
beat_feat_n = librosa.util.normalize(beat_feat, norm=2, axis=0)

# Build recurrence on CQT
rec_width = max(1, min(3, (n_beats - 1) // 2))
R = librosa.segment.recurrence_matrix(beat_feat_n, metric='cosine',
    mode='affinity', sym=True, sparse=False, width=rec_width, axis=-1)
R_enh = librosa.segment.path_enhance(R, n=15, window='hann', clip=True)
W = 9
seq = np.zeros((n_beats, n_beats), dtype=np.float32)
for d in range(1, W+1):
    ww = 1.0 - d/(W+1)
    idx = np.arange(n_beats-d)
    seq[idx, idx+d] = ww; seq[idx+d, idx] = ww
np.fill_diagonal(seq, 1.0)
A = 0.5 * R_enh + 0.5 * seq
A = 0.5 * (A + A.T)
d_vec = A.sum(axis=1)
d_inv_sqrt = np.where(d_vec > 0, 1.0 / np.sqrt(d_vec), 0.0)
L_sym = np.eye(n_beats) - np.diag(d_inv_sqrt) @ A @ np.diag(d_inv_sqrt)
_, evecs = scipy.linalg.eigh(L_sym, subset_by_index=[0, 5])
emb = normalize(evecs[:, 1:6], norm='l2', axis=1)

# Try different numbers of clusters and seeds
for n_c in [3, 4, 5]:
    for seed in [0, 1, 42, 99]:
        km = KMeans(n_clusters=n_c, n_init=50, random_state=seed)
        km.fit(emb)
        labs = km.labels_
        reused = [l for l in np.unique(labs) if np.sum(labs==l) > 0]
        # Check if any label appears in non-contiguous segments
        def count_runs(labels, l):
            return sum(1 for i,lb in enumerate(labels) 
                      if lb == l and (i == 0 or labels[i-1] != l))
        runs = {l: count_runs(list(labs), l) for l in np.unique(labs)}
        multi_run = [l for l, r in runs.items() if r > 1]
        if multi_run:
            print(f'n_c={n_c} seed={seed}: {labs} REUSED={multi_run}')
            break
    else:
        continue
    break
else:
