The transcriber reads a short audio file containing a sequence of synthesized notes, detects where each note begins, estimates its pitch, quantizes the pitch to the nearest MIDI integer, and assigns a velocity proportional to the loudness of each note. import json, sys

AUDIO_DURATION = 5.800   # seconds
TOLERANCE      = 0.1

with open("/workspace/notes.json") as f:
    notes = json.load(f)

assert isinstance(notes, list), "Root must be a JSON array"
assert len(notes) > 0, "Array must be non-empty"

errors = []
prev_onset = -1.0

for i, n in enumerate(notes):
    # Required keys
    for key in ("onset_sec", "offset_sec", "pitch_midi", "velocity"):
        if key not in n:
            errors.append(f"[{i}] missing key '{key}'")

    onset  = n["onset_sec"]
    offset = n["offset_sec"]
    pitch  = n["pitch_midi"]
    vel    = n["velocity"]

    # Types
    if not isinstance(onset,  float): errors.append(f"[{i}] onset_sec not float: {type(onset)}")
    if not isinstance(offset, float): errors.append(f"[{i}] offset_sec not float: {type(offset)}")
    if not isinstance(pitch,  int):   errors.append(f"[{i}] pitch_midi not int: {type(pitch)}")
    if not isinstance(vel,    int):   errors.append(f"[{i}] velocity not int: {type(vel)}")

    # onset < offset (strictly)
    if offset <= onset:
        errors.append(f"[{i}] offset_sec ({offset}) not > onset_sec ({onset})")

    # pitch in [0, 127]
    if not (0 <= pitch <= 127):
        errors.append(f"[{i}] pitch_midi={pitch} out of [0,127]")

    # velocity in [1, 127]
    if not (1 <= vel <= 127):
        errors.append(f"[{i}] velocity={vel} out of [1,127]")

    # offset <= duration + tolerance
    if offset > AUDIO_DURATION + TOLERANCE:
        errors.append(f"[{i}] offset_sec={offset} > duration+tol ({AUDIO_DURATION+TOLERANCE})")

    # strictly ascending onsets
    if onset <= prev_onset:
        errors.append(f"[{i}] onset_sec={onset} not strictly > prev={prev_onset}")
    prev_onset = onset

if errors:
    print("VALIDATION FAILED:")
    for e in errors:
        print(" ", e)
    sys.exit(1)
else:
    print(f"VALIDATION PASSED — {len(notes)} notes, all criteria satisfied.")
    for n in notes:
        print(f"  onset={n['onset_sec']:.4f}  offset={n['offset_sec']:.4f}  "
              f"MIDI={n['pitch_midi']}  vel={n['velocity']}")
