# Manual frequency-domain pitch shift
def freq_domain_pitch_shift(y, sr, n_steps):
    '''Shift pitch by scaling frequency bins in STFT domain'''
    n_fft = 2048
    hop_length = 512
    D = librosa.stft(y, n_fft=n_fft, hop_length=hop_length)
    
    # Scale factor
    factor = 2 ** (n_steps / 12.0)
    n_bins = D.shape[0]
    
    # Create new spectrum by interpolating
    D_shifted = np.zeros_like(D)
    for i in range(n_bins):
        src_bin = i / factor  # source bin
        src_lo = int(src_bin)
        src_hi = src_lo + 1
        frac = src_bin - src_lo
        if src_hi < n_bins:
            D_shifted[i] = (1-frac) * D[src_lo] + frac * D[src_hi]
        elif src_lo < n_bins:
            D_shifted[i] = D[src_lo]
    
    return librosa.istft(D_shifted, hop_length=hop_length, length=len(y))

def spectral_pitch_shift(signal, sr, n_steps, n_fft=2048, hop_length=512):
    '''Shift pitch by interpolating frequency bins in STFT domain.
    This ensures the spectral centroid shifts proportionally.'''
    factor = 2 ** (n_steps / 12.0)
    
    D = librosa.stft(signal, n_fft=n_fft, hop_length=hop_length)
    
    magnitude = np.abs(D)
    phase = np.angle(D)
    
    n_bins = D.shape[0]  # 1 + n_fft//2
    
    # Create shifted magnitude spectrum
    shifted_mag = np.zeros_like(magnitude)
    
    # For each OUTPUT bin, find which INPUT bin contributes to it
    # If we want output_bin to correspond to factor * source_freq,
    # then output_bin = factor * source_bin
    # So source_bin = output_bin / factor
    for out_bin in range(n_bins):
        src_bin_f = out_bin / factor
        src_lo = int(np.floor(src_bin_f))
        src_hi = src_lo + 1
        frac = src_bin_f - src_lo
        
        if src_hi < n_bins:
            shifted_mag[out_bin] = (1 - frac) * magnitude[src_lo] + frac * magnitude[src_hi]
        elif src_lo < n_bins:
            shifted_mag[out_bin] = magnitude[src_lo]
    
    # Reconstruct with random/zero phase (or original phase scaled)
    # Use zero phase for simplicity - this gives musical artifacts but correct centroid
    D_shifted = shifted_mag * np.exp(1j * phase)
    
    return librosa.istft(D_shifted, hop_length=hop_length, length=len(signal))
