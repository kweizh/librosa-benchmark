for i, (on, off) in enumerate(zip(onset_times, offsets)):
    # Find pyin frames in this range
    idx = (pyin_times >= on) & (pyin_times < off)
    voiced_f0 = f0[idx & voiced_flag]

    print(f'Segment {i:2d}: {on:6.3f}s to {off:6.3f}s | Median f0: {median_f0:6.1f} Hz | MIDI: {midi_pitch}')
