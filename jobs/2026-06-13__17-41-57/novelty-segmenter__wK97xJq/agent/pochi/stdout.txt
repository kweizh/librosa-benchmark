import librosa
import numpy as np
from scipy.signal import find_peaks

y, sr = librosa.load('/workspace/input.wav', sr=None)
mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=20, hop_length=512)

# Normalize/standardize MFCCs
mfcc_norm = mfcc - np.mean(mfcc, axis=1, keepdims=True)
mfcc_norm /= np.std(mfcc_norm, axis=1, keepdims=True) + 1e-8

S = librosa.segment.recurrence_matrix(mfcc_norm, mode='affinity', metric='cosine', full=True)

# Let's test a few L values
for L in [20, 30, 40, 50]:
    # Compute novelty curve
    N = S.shape[0]
    sign_X = np.ones((2*L, 2*L))
    sign_X[:, :L] = -1
    sign_Y = np.ones((2*L, 2*L))
    sign_Y[:L, :] = -1
    kernel = sign_X * sign_Y
    
    coords = np.arange(2*L) - (L - 0.5)
    C_X, C_Y = np.meshgrid(coords, coords)
    dist_sq = C_X**2 + C_Y**2
    sigma = L / 2.0
    gaussian = np.exp(-dist_sq / (2 * sigma**2))
    kernel = kernel * gaussian
    
    novelty = np.zeros(N)
    for i in range(L, N - L):
        sub_S = S[i - L : i + L, i - L : i + L]
        novelty[i] = np.sum(sub_S * kernel)
    
    # Normalize novelty curve to [0, 1]
    novelty_min = novelty.min()
    novelty_max = novelty.max()
    if novelty_max > novelty_min:
        novelty_norm = (novelty - novelty_min) / (novelty_max - novelty_min)
    else:
        novelty_norm = novelty
        
    # Find peaks
    # Let's try distance = 1 second (approx 43 frames)
    peaks, properties = find_peaks(novelty_norm, distance=int(sr/512), prominence=0.1)
    peak_times = librosa.frames_to_time(peaks, sr=sr, hop_length=512)
    print(f'L={L}: found {len(peaks)} peaks at times {peak_times}')
