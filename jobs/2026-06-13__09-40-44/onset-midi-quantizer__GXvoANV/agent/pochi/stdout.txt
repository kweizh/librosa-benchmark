import librosa
import numpy as np

y, sr = librosa.load('/workspace/input.wav', sr=None)
duration = librosa.get_duration(y=y, sr=sr)
onsets = librosa.onset.onset_detect(y=y, sr=sr, backtrack=True, units='time')
offsets = list(onsets[1:]) + [duration]

for i, (onset, offset) in enumerate(zip(onsets, offsets)):
    start_sample = int(onset * sr)
    end_sample = int(offset * sr)
    y_seg = y[start_sample:end_sample]
    
    # Run pyin on segment
    try:
        f0, voiced_flag, voiced_probs = librosa.pyin(
            y_seg, 
            fmin=librosa.note_to_hz('C2'), 
            fmax=librosa.note_to_hz('C7'), 
            sr=sr
        )
        voiced_f0 = f0[voiced_flag]
        if len(voiced_f0) > 0:
            median_f0 = np.median(voiced_f0)
            midi = int(round(librosa.hz_to_midi(median_f0)))
        else:
            median_f0 = None
            midi = None
        print(f'Seg {i}: {onset:.3f} - {offset:.3f} | len={len(y_seg)} | median_f0={median_f0} | midi={midi}')
    except Exception as e:
        print(f'Seg {i} failed: {e}')

import librosa
import numpy as np

y, sr = librosa.load('/workspace/input.wav', sr=None)
duration = librosa.get_duration(y=y, sr=sr)
onsets = librosa.onset.onset_detect(y=y, sr=sr, backtrack=True, units='time')
offsets = list(onsets[1:]) + [duration]

hop_length = 512
f0, voiced_flag, voiced_probs = librosa.pyin(
    y, 
    fmin=librosa.note_to_hz('C2'), 
    fmax=librosa.note_to_hz('C7'), 
    sr=sr,
    hop_length=hop_length
)
times = librosa.frames_to_time(np.arange(len(f0)), sr=sr, hop_length=hop_length)

raw_notes = []
for i, (onset, offset) in enumerate(zip(onsets, offsets)):
    # Compute RMS energy of this segment
    start_sample = int(onset * sr)
    end_sample = int(offset * sr)
    y_seg = y[start_sample:end_sample]
    if len(y_seg) == 0:
        continue
    rms = np.sqrt(np.mean(y_seg**2))
    
    # Filter by RMS energy
    if rms < 0.05:
        continue
        
    # Find voiced f0 frames in this segment
    mask = (times >= onset) & (times < offset)
    voiced_in_seg = voiced_flag[mask]
    f0_in_seg = f0[mask]
    voiced_f0 = f0_in_seg[voiced_in_seg]
    
    if len(voiced_f0) > 0:
        median_f0 = np.median(voiced_f0)
        midi = int(round(librosa.hz_to_midi(median_f0)))
        raw_notes.append({
            'onset_sec': float(onset),
            'offset_sec': float(offset),
            'pitch_midi': midi,
            'rms': float(rms)
        })

# Merge consecutive notes with the same pitch
merged_notes = []
for note in raw_notes:
    if not merged_notes:
        merged_notes.append(note)
    else:
        last_note = merged_notes[-1]
        # Check if they are consecutive and have the same pitch
        # Since we skipped silent segments, if they are adjacent in raw_notes,
        # and the gap between them is very small, we can merge them.
        # Let's check if the gap between last_note['offset_sec'] and note['onset_sec'] is small (e.g. < 0.05s)
        gap = note['onset_sec'] - last_note['offset_sec']
        if last_note['pitch_midi'] == note['pitch_midi'] and gap < 0.05:
            # Merge: update offset and average RMS or keep max RMS
            last_note['offset_sec'] = note['offset_sec']
            # We can update RMS as weighted average or just max
            last_note['rms'] = max(last_note['rms'], note['rms'])
        else:
            merged_notes.append(note)

for i, note in enumerate(merged_notes):
    print(f'Note {i}: {note}')
