import numpy as np
import librosa
import librosa.effects

y, sr = librosa.load("/workspace/input.wav", sr=None, mono=True)
harmonic, percussive = librosa.effects.hpss(y, margin=(1.0, 5.0))

harmonic_shifted = librosa.effects.pitch_shift(harmonic, sr=sr, n_steps=7)

original_length = len(y)

def align(arr, n):
    return arr[:n] if len(arr) >= n else np.pad(arr, (0, n - len(arr)))

# centroid of each component
c_in     = np.mean(librosa.feature.spectral_centroid(y=y, sr=sr))
c_harm   = np.mean(librosa.feature.spectral_centroid(y=harmonic, sr=sr))
c_hshift = np.mean(librosa.feature.spectral_centroid(y=harmonic_shifted, sr=sr))
c_perc   = np.mean(librosa.feature.spectral_centroid(y=percussive, sr=sr))

print(f"input centroid        : {c_in:.1f}")
print(f"harmonic centroid     : {c_harm:.1f}")
print(f"harmonic+7st centroid : {c_hshift:.1f}")
print(f"percussive centroid   : {c_perc:.1f}")

# energy
print(f"\nharmonic RMS  : {np.sqrt(np.mean(harmonic**2)):.4f}")
print(f"percussive RMS: {np.sqrt(np.mean(percussive**2)):.4f}")
