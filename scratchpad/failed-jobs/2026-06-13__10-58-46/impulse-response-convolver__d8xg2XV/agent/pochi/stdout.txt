import numpy as np
import soundfile as sf
import librosa
from scipy.signal import fftconvolve

# Read output
wet, wet_sr = sf.read('/workspace/wet.wav')
print(f'Output sr: {wet_sr}, len: {len(wet)}')
print(f'Output peak: {np.max(np.abs(wet)):.6f}')

# Read inputs
dry, dry_sr = librosa.load('/workspace/dry.wav', sr=None, mono=True)
ir, ir_sr = librosa.load('/workspace/ir.wav', sr=None, mono=True)

# Resample IR
ir_resampled = librosa.resample(ir, orig_sr=ir_sr, target_sr=dry_sr)
print(f'Expected len: {len(dry) + len(ir_resampled) - 1}')
print(f'Length diff: {abs(len(wet) - (len(dry) + len(ir_resampled) - 1))}')

# Compute reference
ref = fftconvolve(dry, ir_resampled)
peak_ref = np.max(np.abs(ref))
ref = ref * (0.95 / peak_ref)

# Cosine similarity
cos_sim = np.dot(wet, ref) / (np.linalg.norm(wet) * np.linalg.norm(ref))
print(f'Cosine similarity: {cos_sim:.10f}')
print(f'All checks passed: sr_match={wet_sr == dry_sr}, len_ok={abs(len(wet) - (len(dry) + len(ir_resampled) - 1)) <= 1}, peak_ok={0.90 <= np.max(np.abs(wet)) <= 1.0}, cos_sim_ok={cos_sim >= 0.99}')
