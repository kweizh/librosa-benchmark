import numpy as np
import librosa
import soundfile as sf

SR, HL = 22050, 512
ref, _ = sf.read('/workspace/reference.wav')
perf, _ = sf.read('/workspace/performance.wav')
if ref.ndim > 1: ref = ref.mean(axis=1)
if perf.ndim > 1: perf = perf.mean(axis=1)

cr = librosa.feature.chroma_cens(y=ref, sr=SR, hop_length=HL)
cp = librosa.feature.chroma_cens(y=perf, sr=SR, hop_length=HL)

print('ref min/max:', cr.min(), cr.max())
print('ref nans:', np.isnan(cr).sum(), 'zeros:', (cr == 0).sum(), 'total:', cr.size)
print('perf min/max:', cp.min(), cp.max())
print('perf nans:', np.isnan(cp).sum(), 'zeros:', (cp == 0).sum(), 'total:', cp.size)
