print(f'Sample rate: {info.samplerate}')
print(f'Channels: {info.channels}')
print(f'Frames: {info.frames}')
print(f'Duration: {info.duration:.3f}s')

# Read the waveform
y, sr = sf.read('reconstructed.wav')
print(f'Waveform shape: {y.shape}')
print(f'Peak amplitude: {np.max(np.abs(y)):.6f}')
print(f'In range [0.01, 1.0]: {0.01 <= np.max(np.abs(y)) <= 1.0}')

# Verify magnitude match
mag_orig = np.load('magnitude.npy')
n_bins = mag_orig.shape[0]
n_fft = 2 * (n_bins - 1)
hop_length = n_fft // 4

D_reconstructed = librosa.stft(y, n_fft=n_fft, hop_length=hop_length, win_length=n_fft, window='hann', center=True)
mag_reconstructed = np.abs(D_reconstructed)

# Check shape match
print(f'Original mag shape: {mag_orig.shape}')
print(f'Reconstructed mag shape: {mag_reconstructed.shape}')

# Compute relative error
diff = np.abs(mag_orig - mag_reconstructed)
rel_error = np.mean(diff) / np.mean(mag_orig)
print(f'Mean relative magnitude error: {rel_error:.6f}')

# Check length is close (within a few hop lengths)
expected_len = mag_orig.shape[1] * hop_length
print(f'Expected length: {expected_len}')
print(f'Actual length: {len(y)}')
print(f'Difference: {abs(len(y) - expected_len)} (within {hop_length * 4} = {4 * hop_length}? {abs(len(y) - expected_len) <= 4 * hop_length})')

print()
