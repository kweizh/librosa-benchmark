for r in rows:
    dur = float(r['end_time_sec']) - float(r['start_time_sec'])
    rms_val = float(r['rms'])
    total_weighted_sq += dur * (rms_val ** 2)
    total_duration += dur

streaming_rms = math.sqrt(total_weighted_sq / total_duration)
