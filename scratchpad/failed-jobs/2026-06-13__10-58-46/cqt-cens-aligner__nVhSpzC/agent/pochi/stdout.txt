import numpy as np
import librosa
import soundfile as sf

y, sr = sf.read('/workspace/input.wav')
if y.ndim > 1:
    y = y[:, 0]

hop_length = 512
n_bins = 84

C = np.abs(librosa.cqt(y, sr=sr, hop_length=hop_length, n_bins=n_bins, bins_per_octave=12))
C_db = librosa.amplitude_to_db(C, ref=np.max)

print(f'C_db shape: {C_db.shape}')
print(f'C_db min: {C_db.min():.4f}')
print(f'C_db max: {C_db.max():.4f}')
print(f'C_db NaN: {np.isnan(C_db).sum()}')
print(f'C_db Inf: {np.isinf(C_db).sum()}')

# Check: the max should be 0.0
print(f'Any element at exactly 0 dB? {(C_db == 0.0).sum()}')

# After beat sync
tempo, beat_frames = librosa.beat.beat_track(y=y, sr=sr, hop_length=hop_length)
C_sync = librosa.util.sync(C_db, beat_frames, aggregate=np.median)
print(f'C_sync max: {C_sync.max():.4f}')
