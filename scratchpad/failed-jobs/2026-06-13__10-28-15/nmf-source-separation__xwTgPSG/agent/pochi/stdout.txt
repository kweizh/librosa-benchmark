import numpy as np
import librosa
from sklearn.decomposition import NMF

y, sr = librosa.load("/workspace/input.wav", sr=None, mono=True)
D = librosa.stft(y, n_fft=2048, hop_length=512)
S = np.abs(D)

# Try Wiener masking approach:
# mask_k = S_k / (sum of all S_k + eps)  -- soft Wiener mask applied to original D
nmf = NMF(n_components=4, init="nndsvda", max_iter=500, random_state=42)
W = nmf.fit_transform(S)
H = nmf.components_
S_recon = W @ H

# Wiener masks
eps = 1e-8
masks = []
for k in range(4):
    S_k = np.outer(W[:, k], H[k, :])
    mask = S_k / (S_recon + eps)
    masks.append(mask)

# Each component = mask * D (complex)
components = []
for k in range(4):
    y_k = librosa.istft(masks[k] * D, hop_length=512, length=len(y))
    components.append(y_k)

mixture = np.sum(components, axis=0)
cos = np.dot(mixture, y) / (np.linalg.norm(mixture)*np.linalg.norm(y))
print(f"Wiener mask cosine similarity: {cos:.4f}")

    centroid = librosa.feature.spectral_centroid(y=c, sr=sr).mean()
