import json
import librosa
import numpy as np

# Load the JSON
with open('beats.json') as f:
    data = json.load(f)

print('Keys:', list(data.keys()))
print('global_tempo_bpm:', data['global_tempo_bpm'])
print('Number of beats:', len(data['beat_times_sec']))
print()

# Validate schema
assert 'global_tempo_bpm' in data, 'Missing global_tempo_bpm'
assert 'beat_times_sec' in data, 'Missing beat_times_sec'
assert len(data) == 2, f'Expected 2 keys, got {len(data)}'

# Validate global_tempo_bpm is a finite number in [30, 300]
tempo = data['global_tempo_bpm']
assert isinstance(tempo, (int, float)), f'tempo is {type(tempo)}'
assert np.isfinite(tempo), 'tempo is not finite'
assert 30 <= tempo <= 300, f'tempo {tempo} out of range [30, 300]'

# Validate beat_times_sec
beats = data['beat_times_sec']
assert len(beats) > 0, 'beat_times_sec is empty'
assert all(isinstance(b, (int, float)) for b in beats), 'non-numeric beat times'

# Check strictly increasing
for i in range(1, len(beats)):
    assert beats[i] > beats[i-1], f'beats not strictly increasing at index {i}: {beats[i-1]} >= {beats[i]}'

# Check all within [0, audio_duration + 0.5]
y, sr = librosa.load('input.wav', sr=None)
duration = len(y) / sr
max_beat = max(beats)
assert max_beat <= duration + 0.5, f'max beat {max_beat} > duration+0.5 {duration+0.5}'
assert min(beats) >= 0, f'min beat {min(beats)} < 0'

# Check median inter-beat interval corresponds to tempo within 15% of global_tempo_bpm
ibis = [beats[i] - beats[i-1] for i in range(1, len(beats))]
median_ibi = np.median(ibis)
median_tempo = 60.0 / median_ibi
print(f'Median IBI: {median_ibi:.4f}s -> Median tempo: {median_tempo:.2f} BPM')
print(f'Global tempo: {tempo:.2f} BPM')
ratio = median_tempo / tempo
print(f'Ratio (median_tempo / global_tempo): {ratio:.4f}')
assert 0.85 <= ratio <= 1.15, f'Median tempo {median_tempo:.2f} not within 15% of global tempo {tempo:.2f}'

print()
print('ALL VALIDATION CHECKS PASSED!')
