import json
import os

# Load results
with open('results.json') as f:
    data = json.load(f)

# Check schema
assert 'results' in data, 'Missing results key'
results = data['results']

# Check count matches library
lib_files = [f for f in os.listdir('library') if f.endswith('.wav')]
print(f'Library files: {len(lib_files)}')
print(f'Result entries: {len(results)}')
assert len(results) == len(lib_files), 'Count mismatch'

# Check filenames are basenames
for r in results:
    fname = r['filename']
    assert fname == os.path.basename(fname), f'{fname} is not a basename'
    assert fname in lib_files, f'{fname} not in library'
    sim = r['similarity']
    assert -1.0 <= sim <= 1.0, f'{sim} out of range'
    assert isinstance(sim, float) and json.dumps(sim) != 'Infinity' and json.dumps(sim) != 'NaN', f'{sim} not finite'

# Check no duplicates
fnames = [r['filename'] for r in results]
assert len(fnames) == len(set(fnames)), 'Duplicate filenames'

# Check sorted descending
sims = [r['similarity'] for r in results]
for i in range(len(sims)-1):
    assert sims[i] >= sims[i+1], f'Not sorted: {sims[i]} < {sims[i+1]}'

# Check ground truth match
print(f\"Top match: {results[0]['filename']} (similarity: {results[0]['similarity']:.4f})\")

    gt = f.read().strip()
