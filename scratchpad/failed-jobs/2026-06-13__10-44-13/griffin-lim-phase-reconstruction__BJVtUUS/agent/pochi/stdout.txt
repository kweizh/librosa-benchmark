import numpy as np
import librosa
import soundfile as sf

# Load magnitude spectrogram
mag = np.load('magnitude.npy')
print('Input shape:', mag.shape)

# Infer STFT parameters
n_bins = mag.shape[0]
n_fft = 2 * (n_bins - 1)  # 1024
hop_length = n_fft // 4    # 256 (librosa default)
sr = 22050

print(f'n_fft={n_fft}, hop_length={hop_length}, sr={sr}')

# Reconstruct using Griffin-Lim with enough iterations
waveform = librosa.griffinlim(mag, n_iter=100, hop_length=hop_length, n_fft=n_fft)

print('Waveform shape:', waveform.shape)
print('Waveform dtype:', waveform.dtype)
print('Peak absolute amplitude:', np.max(np.abs(waveform)))
print('Waveform length in samples:', len(waveform))
print('Duration:', len(waveform) / sr, 'seconds')

# Save as WAV
sf.write('reconstructed.wav', waveform, sr)
print('Saved reconstructed.wav')

# Verify: recompute magnitude STFT and compare
S_recon = np.abs(librosa.stft(waveform, n_fft=n_fft, hop_length=hop_length))
print('Reconstructed magnitude shape:', S_recon.shape)

# Compute error metrics
mse = np.mean((mag - S_recon)**2)
print('MSE between original and reconstructed magnitude:', mse)

# Check shapes match
if mag.shape == S_recon.shape:
    print('Shapes match exactly')
else:
    print(f'Shape mismatch: original {mag.shape} vs reconstructed {S_recon.shape}')
    # Compare on overlapping frames
    min_frames = min(mag.shape[1], S_recon.shape[1])
    mse_overlap = np.mean((mag[:, :min_frames] - S_recon[:, :min_frames])**2)
    print(f'MSE on overlapping frames: {mse_overlap}')
