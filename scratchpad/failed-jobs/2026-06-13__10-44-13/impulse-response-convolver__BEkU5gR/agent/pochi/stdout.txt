import soundfile as sf
import numpy as np

wet, sr_wet = sf.read('/workspace/wet.wav')
dry, sr_dry = sf.read('/workspace/dry.wav')
ir, sr_ir = sf.read('/workspace/ir.wav')

import librosa
ir_resampled = librosa.resample(ir[:,0] if ir.ndim > 1 else ir, orig_sr=sr_ir, target_sr=sr_dry)
expected_len = len(dry) + len(ir_resampled) - 1

print(f'wet.wav: sr={sr_wet}, samples={len(wet)}, peak={np.max(np.abs(wet)):.6f}')

# Cosine similarity with reference convolution
from scipy.signal import fftconvolve
dry_mono = dry[:,0] if dry.ndim > 1 else dry
ref = fftconvolve(dry_mono, ir_resampled, mode='full')
ref_peak = np.max(np.abs(ref))
if ref_peak > 0:
    ref = ref * (0.95 / ref_peak)
wet_mono = wet[:,0] if wet.ndim > 1 else wet
cos_sim = np.dot(wet_mono, ref) / (np.linalg.norm(wet_mono) * np.linalg.norm(ref))
print(f'Cosine similarity: {cos_sim:.6f}')
print(f'Cosine sim >= 0.99: {cos_sim >= 0.99}')
