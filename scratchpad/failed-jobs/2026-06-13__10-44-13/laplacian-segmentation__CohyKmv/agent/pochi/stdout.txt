import json
import librosa

# Load segments
with open('/workspace/segments.json') as f:
    segments = json.load(f)

# Load audio to get duration
y, sr = librosa.load('/workspace/input.wav', sr=None)
duration = librosa.get_duration(y=y, sr=sr)

print(f'Audio duration: {duration:.2f}s')
print(f'Number of segments: {len(segments)}')
print()

# Check all criteria
errors = []

# 1. Valid JSON list of objects with 'start' and 'end' float keys
if not isinstance(segments, list):
    errors.append('Not a list')
for s in segments:
    if not isinstance(s, dict) or 'start' not in s or 'end' not in s:
        errors.append(f'Bad segment: {s}')
    if not isinstance(s['start'], (int, float)) or not isinstance(s['end'], (int, float)):
        errors.append(f'Non-float values: {s}')

# 2. Sorted by start
for i in range(1, len(segments)):
    if segments[i]['start'] < segments[i-1]['start']:
        errors.append(f'Not sorted at index {i}')

# 3. No negative durations
for i, s in enumerate(segments):
    dur = s['end'] - s['start']
    if dur < 0:
        errors.append(f'Negative duration at segment {i}: {dur}')
    if dur <= 1.0:
        errors.append(f'Duration <= 1.0s at segment {i}: {dur:.4f}s')

# 4. Coverage: first starts ~0, last ends ~duration
if segments[0]['start'] > 0.5:
    errors.append(f'First segment start too late: {segments[0][\"start\"]}')
if abs(segments[-1]['end'] - duration) > 0.5:
    errors.append(f'Last segment end mismatch: {segments[-1][\"end\"]} vs {duration:.2f}')

# 5. Contiguity: no gap > 0.5s, no overlap > 0.1s
for i in range(1, len(segments)):
    gap = segments[i]['start'] - segments[i-1]['end']
    if gap > 0.5:
        errors.append(f'Gap > 0.5s between segments {i-1} and {i}: {gap:.4f}s')
    if gap < -0.1:
        errors.append(f'Overlap > 0.1s between segments {i-1} and {i}: {gap:.4f}s')

# 6. Total segments between 3 and 15
if len(segments) < 3 or len(segments) > 15:
    errors.append(f'Segment count out of range: {len(segments)}')

if errors:
    print('ERRORS:')
    for e in errors:
        print(f'  - {e}')
else:
    print('ALL CHECKS PASSED!')
    
print()
for i, s in enumerate(segments):
    print(f'  Seg {i}: {s[\"start\"]:.2f} - {s[\"end\"]:.2f} ({s[\"end\"]-s[\"start\"]:.2f}s)')

**Pipeline summary:**
1. **Beat tracking** – `librosa.beat.beat_track` extracted 213 beat positions from the ~119.88s audio file.
2. **Beat-synchronous spectral features** – CQT chroma features were computed and synced to beat frames via `librosa.util.sync` with median aggregation, then L2-normalised.
3. **Recurrence/affinity matrix** – A cosine-similarity k-nearest-neighbour graph (k=15) was built on the normalised chroma vectors and symmetrised.
4. **Normalised graph Laplacian** – `scipy.sparse.csgraph.laplacian(W, normed=True)` produced the normalised Laplacian.
5. **Eigendecomposition** – `numpy.linalg.eigh` decomposed the Laplacian; the first 5 non-trivial eigenvectors (indices 1–5) formed the spectral embedding.
6. **Agglomerative clustering** – `sklearn.cluster.AgglomerativeClustering` with Ward linkage on the row-normalised embedding produced 8 cluster labels, which yielded 15 contiguous segments after boundary extraction.
