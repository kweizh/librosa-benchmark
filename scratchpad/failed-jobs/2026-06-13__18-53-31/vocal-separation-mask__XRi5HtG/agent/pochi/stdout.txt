import soundfile as sf
import numpy as np

y_in, sr_in = sf.read('input.wav')
y_v, sr_v = sf.read('vocal.wav')
y_a, sr_a = sf.read('accompaniment.wav')

print(f'Input:  shape={y_in.shape}, sr={sr_in}, dtype={y_in.dtype}')
print(f'Vocal:  shape={y_v.shape}, sr={sr_v}, dtype={y_v.dtype}')
print(f'Accomp: shape={y_a.shape}, sr={sr_a}, dtype={y_a.dtype}')

# Check sample rates match
assert sr_v == sr_in, f'Sample rate mismatch: vocal {sr_v} vs input {sr_in}'
assert sr_a == sr_in, f'Sample rate mismatch: accomp {sr_a} vs input {sr_in}'

# Check lengths within ±1
assert abs(len(y_v) - len(y_in)) <= 1, f'Vocal length off: {len(y_v)} vs {len(y_in)}'
assert abs(len(y_a) - len(y_in)) <= 1, f'Accomp length off: {len(y_a)} vs {len(y_in)}'

# Check non-silent
rms_v = np.sqrt(np.mean(y_v**2))
rms_a = np.sqrt(np.mean(y_a**2))
assert rms_v > 1e-4, f'Vocal RMS too low: {rms_v}'
assert rms_a > 1e-4, f'Accomp RMS too low: {rms_a}'

# Check reconstruction MAE
min_len = min(len(y_v), len(y_a), len(y_in))
mae = np.mean(np.abs(y_v[:min_len] + y_a[:min_len] - y_in[:min_len]))
assert mae <= 5e-2, f'MAE too high: {mae}'

# Check not bitwise copies
assert not np.array_equal(y_v[:min_len], y_in[:min_len]), 'Vocal is a bitwise copy of input!'
assert not np.array_equal(y_a[:min_len], y_in[:min_len]), 'Accompaniment is a bitwise copy of input!'

print('All acceptance criteria PASSED!')
