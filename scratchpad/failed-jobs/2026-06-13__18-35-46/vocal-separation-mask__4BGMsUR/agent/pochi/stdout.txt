import librosa
import numpy as np
import soundfile as sf
import sys

def main():
    input_path = '/workspace/input.wav'
    y, sr = librosa.load(input_path, sr=None, mono=True)
    
    # Compute complex STFT
    D = librosa.stft(y)
    S_full = np.abs(D)
    
    # Estimate background using nearest-neighbor median filtering
    # width constraint prevents neighbors that are too close in time
    width = int(librosa.time_to_frames(2, sr=sr))
    S_filter = librosa.decompose.nn_filter(S_full,
                                           aggregate=np.median,
                                           metric='cosine',
                                           width=width)
    
    # The background should not exceed the original magnitude
    S_filter = np.minimum(S_full, S_filter)
    
    # The foreground is the difference
    S_foreground = S_full - S_filter
    
    # Derive complementary soft masks
    mask_bg = librosa.util.softmask(S_filter, S_foreground, power=2)
    mask_fg = librosa.util.softmask(S_foreground, S_filter, power=2)
    
    # Apply masks to the original complex STFT
    D_bg = D * mask_bg
    D_fg = D * mask_fg
    
    # Reconstruct waveforms
    y_bg = librosa.istft(D_bg, length=len(y))
    y_fg = librosa.istft(D_fg, length=len(y))
    
    # Save outputs
    sf.write('/workspace/accompaniment.wav', y_bg, sr)
    sf.write('/workspace/vocal.wav', y_fg, sr)

if __name__ == '__main__':
    main()
