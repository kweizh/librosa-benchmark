import librosa
import numpy as np
import soundfile as sf
import json

def main():
    y, sr = librosa.load('/workspace/mixture.wav', sr=None)
    
    D = librosa.stft(y)
    S = np.abs(D)
    
    n_components = 4
    W, H = librosa.decompose.decompose(S, n_components=n_components, sort=True)
    
    # Reconstruct components
    S_comps = []
    for k in range(n_components):
        S_k = W[:, k:k+1] @ H[k:k+1, :]
        S_comps.append(S_k)
        
    S_comps = np.array(S_comps)
    
    # Wiener-style soft mask
    S_sq = S_comps ** 2
    S_sq_sum = np.sum(S_sq, axis=0) + 1e-10
    
    components_info = []
    
    for k in range(n_components):
        M_k = S_sq[k] / S_sq_sum
        D_k = M_k * D
        y_k = librosa.istft(D_k, length=len(y))
        
        sf.write(f'/workspace/component_{k}.wav', y_k, sr)
        
        centroid = librosa.feature.spectral_centroid(y=y_k, sr=sr)
        mean_centroid = float(np.mean(centroid))
        
        rms = float(np.sqrt(np.mean(y_k**2)))
        
        components_info.append({
            "index": k,
            "centroid_hz": mean_centroid,
            "rms": rms
        })
        
    with open('/workspace/components.json', 'w') as f:
        json.dump(components_info, f, indent=2)

if __name__ == "__main__":
    main()
