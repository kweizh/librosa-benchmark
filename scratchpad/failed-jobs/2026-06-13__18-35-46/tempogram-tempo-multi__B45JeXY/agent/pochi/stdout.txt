import librosa
import numpy as np
import json
from scipy.signal import find_peaks

y, sr = librosa.load('/workspace/input.wav', sr=None, mono=True)

hop_length = 512
onset_env = librosa.onset.onset_strength(y=y, sr=sr, hop_length=hop_length)

tempogram = librosa.feature.fourier_tempogram(onset_envelope=onset_env, sr=sr, hop_length=hop_length)
salience = np.mean(np.abs(tempogram), axis=1)

win_length = 384
tempo_freqs = librosa.fourier_tempo_frequencies(sr=sr, hop_length=hop_length, win_length=win_length)
tempo_bpms = tempo_freqs * 60.0

valid_idx = np.where((tempo_bpms >= 40.0) & (tempo_bpms <= 240.0))[0]

valid_bpms = tempo_bpms[valid_idx]
valid_salience = salience[valid_idx]

peaks, _ = find_peaks(valid_salience)

if len(peaks) < 3:
    peaks = np.argsort(valid_salience)[::-1][:3]

peak_bpms = valid_bpms[peaks]
peak_saliences = valid_salience[peaks]

sort_idx = np.argsort(peak_saliences)[::-1]
top_peaks = sort_idx[:3]

top_bpms = peak_bpms[top_peaks]
top_saliences = peak_saliences[top_peaks]

ref_tempo = librosa.feature.tempo(onset_envelope=onset_env, sr=sr, hop_length=hop_length, aggregate=np.mean)[0]

beat_tempo, _ = librosa.beat.beat_track(onset_envelope=onset_env, sr=sr, hop_length=hop_length)
if isinstance(beat_tempo, np.ndarray):
    beat_tempo = beat_tempo[0]

print("Ref tempo:", ref_tempo)
print("Beat tempo:", beat_tempo)
print("Top BPMs:", top_bpms)

# Form JSON
out = []
for i in range(3):
    out.append({
        "tempo_bpm": float(top_bpms[i]),
        "salience": float(top_saliences[i]),
        "harmonic_rank": i + 1
    })

with open('/workspace/tempo_candidates.json', 'w') as f:
    json.dump(out, f, indent=2)
