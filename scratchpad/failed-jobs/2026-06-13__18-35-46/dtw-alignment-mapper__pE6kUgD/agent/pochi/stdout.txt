import librosa
import json
import numpy as np

def main():
    sr = 22050
    hop_length = 512

    # Load audio
    ref_y, _ = librosa.load('/workspace/reference.wav', sr=sr)
    comp_y, _ = librosa.load('/workspace/comparison.wav', sr=sr)

    # Compute CENS chroma
    chroma_ref = librosa.feature.chroma_cens(y=ref_y, sr=sr, hop_length=hop_length)
    chroma_comp = librosa.feature.chroma_cens(y=comp_y, sr=sr, hop_length=hop_length)

    # Run DTW
    D, wp = librosa.sequence.dtw(X=chroma_ref, Y=chroma_comp, metric='cosine')

    # wp is reversed
    wp = wp[::-1]

    # total cost
    total_cost = float(D[-1, -1])

    # warping path frames
    warping_path_frames = wp.tolist()

    # generate timestamp map
    step = 21 # 21 * 512 / 22050 = 0.4876 seconds
    max_ref_frame = wp[-1, 0]
    target_frames = list(range(0, max_ref_frame + 1, step))

    timestamp_map = []
    for trf in target_frames:
        idx = np.searchsorted(wp[:, 0], trf)
        if idx >= len(wp):
            idx = len(wp) - 1
        
        comp_frame = wp[idx, 1]
        
        ref_time = float(librosa.frames_to_time(frames=trf, sr=sr, hop_length=hop_length))
        comp_time = float(librosa.frames_to_time(frames=comp_frame, sr=sr, hop_length=hop_length))
        
        timestamp_map.append({
            "ref_time": ref_time,
            "comp_time": comp_time
        })

    # Save to json
    out = {
        "warping_path_frames": warping_path_frames,
        "timestamp_map": timestamp_map,
        "total_cost": total_cost,
        "hop_length": hop_length,
        "sample_rate": sr
    }

    with open('/workspace/alignment.json', 'w') as f:
        json.dump(out, f, indent=2)

if __name__ == "__main__":
    main()
