import sys
import json
import librosa
import numpy as np
import scipy
from sklearn.cluster import AgglomerativeClustering

def process():
    audio_path = '/workspace/input.wav'
    output_path = '/workspace/segments.json'
    
    y, sr = librosa.load(audio_path)
    duration = librosa.get_duration(y=y, sr=sr)
    
    # Feature extraction
    chroma = librosa.feature.chroma_cqt(y=y, sr=sr)
    mfcc = librosa.feature.mfcc(y=y, sr=sr)
    
    # Beat tracking
    tempo, beats = librosa.beat.beat_track(y=y, sr=sr)
    beats = librosa.util.fix_frames(beats, x_min=0)
    
    # Beat synchronization
    chroma_sync = librosa.util.sync(chroma, beats, aggregate=np.median)
    mfcc_sync = librosa.util.sync(mfcc, beats, aggregate=np.median)
    
    features = np.vstack([chroma_sync, mfcc_sync])
    
    # Recurrence matrix
    R = librosa.segment.recurrence_matrix(features, mode='affinity', sym=True)
    R = librosa.segment.path_enhance(R, 15)
    
    # Normalized Laplacian
    L = scipy.sparse.csgraph.laplacian(R, normed=True)
    
    # Eigendecomposition
    evals, evecs = scipy.linalg.eigh(L)
    
    k = min(8, evecs.shape[1] - 1)
    if k < 1:
        k = 1
    X = evecs[:, 1:k+1]
    
    # Agglomerative clustering
    clusterer = AgglomerativeClustering(n_clusters=k)
    labels = clusterer.fit_predict(X)
    
    # Median filter to smooth labels
    labels = scipy.signal.medfilt(labels, kernel_size=9)
    
    # Find boundaries
    bounds = [0]
    for i in range(1, len(labels)):
        if labels[i] != labels[i-1]:
            bounds.append(i)
    bounds.append(len(labels))
    
    beat_times = librosa.frames_to_time(beats, sr=sr)
    
    segments = []
    for i in range(len(bounds) - 1):
        start_idx = bounds[i]
        end_idx = bounds[i+1]
        
        start_time = beat_times[start_idx] if start_idx < len(beat_times) else duration
        end_time = beat_times[end_idx] if end_idx < len(beat_times) else duration
        
        if i == len(bounds) - 2:
            end_time = duration
            
        segments.append({
            "start": float(start_time),
            "end": float(end_time)
        })
        
    # Enforce duration > 1.0
    merged = True
    while merged:
        merged = False
        for i in range(len(segments)):
            if segments[i]['end'] - segments[i]['start'] < 1.0:
                if i > 0:
                    segments[i-1]['end'] = segments[i]['end']
                    segments.pop(i)
                    merged = True
                    break
                elif i < len(segments) - 1:
                    segments[i+1]['start'] = segments[i]['start']
                    segments.pop(i)
                    merged = True
                    break

    # Enforce number of segments between 3 and 15
    while len(segments) > 15:
        shortest_idx = min(range(len(segments)), key=lambda i: segments[i]['end'] - segments[i]['start'])
        if shortest_idx > 0:
            segments[shortest_idx-1]['end'] = segments[shortest_idx]['end']
            segments.pop(shortest_idx)
        else:
            segments[shortest_idx+1]['start'] = segments[shortest_idx]['start']
            segments.pop(shortest_idx)
            
    while len(segments) < 3:
        longest_idx = max(range(len(segments)), key=lambda i: segments[i]['end'] - segments[i]['start'])
        mid = (segments[longest_idx]['start'] + segments[longest_idx]['end']) / 2
        new_seg = {"start": mid, "end": segments[longest_idx]['end']}
        segments[longest_idx]['end'] = mid
        segments.insert(longest_idx + 1, new_seg)
        
    segments[0]['start'] = 0.0
    segments[-1]['end'] = duration
    for i in range(len(segments) - 1):
        segments[i+1]['start'] = segments[i]['end']
        
    with open(output_path, 'w') as f:
        json.dump(segments, f, indent=2)

if __name__ == '__main__':
    process()
