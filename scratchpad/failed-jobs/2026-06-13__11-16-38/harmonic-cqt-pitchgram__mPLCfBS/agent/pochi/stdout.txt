import librosa
import numpy as np

def main():
    # Load audio
    y, sr = librosa.load('/workspace/input.wav', sr=None)

    harmonics = [0.5, 1, 2, 3, 4, 5]
    bins_per_octave = 12
    n_bins = 72
    base_fmin = librosa.note_to_hz('C1')

    hcqt_list = []

    for h in harmonics:
        fmin_h = base_fmin * h
        
        # Compute CQT
        C = librosa.cqt(y=y, sr=sr, fmin=fmin_h, n_bins=n_bins, bins_per_octave=bins_per_octave)
        
        # Magnitude
        C_mag = np.abs(C)
        
        # Convert to dB
        C_db = librosa.amplitude_to_db(C_mag, ref=np.max)
        
        hcqt_list.append(C_db)

    # Stack to (6, 72, n_frames)
    hcqt = np.stack(hcqt_list, axis=0).astype(np.float32)

    # Save to /workspace/hcqt.npy
    np.save('/workspace/hcqt.npy', hcqt)

if __name__ == '__main__':
    main()
