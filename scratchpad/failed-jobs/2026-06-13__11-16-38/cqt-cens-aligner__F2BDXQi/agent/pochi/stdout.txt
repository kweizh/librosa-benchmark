import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
import librosa
import librosa.display
import numpy as np

# Load audio
y, sr = librosa.load('/workspace/input.wav')

# Compute CQT
hop_length = 512
C = np.abs(librosa.cqt(y, sr=sr, hop_length=hop_length))
C_db = librosa.amplitude_to_db(C, ref=np.max)

# Track beats
tempo, beats = librosa.beat.beat_track(y=y, sr=sr, hop_length=hop_length)

# Aggregate CQT per beat
C_sync = librosa.util.sync(C_db, beats, aggregate=np.median)
C_sync = C_sync.astype(np.float32)

# Save numpy array
np.save('/workspace/cqt_db.npy', C_sync)

# Plot and save image
fig, ax = plt.subplots()

# Sync adds a column sometimes, beat_times might not match C_sync shape
beat_times = librosa.frames_to_time(librosa.util.fix_frames(beats, x_min=0, x_max=C.shape[1]), sr=sr, hop_length=hop_length)
img = librosa.display.specshow(C_sync, y_axis='cqt_note', x_axis='time', 
                               x_coords=beat_times, sr=sr, hop_length=hop_length, ax=ax)
fig.colorbar(img, ax=ax, format="%+2.0f dB")
ax.set_title('Beat-synchronous CQT')
plt.savefig('/workspace/cqt.png', dpi=150)
plt.close()
