import numpy as np
import librosa
import soundfile as sf

def main():
    # Load the magnitude spectrogram
    S = np.load('/workspace/magnitude.npy')
    
    # Infer n_fft
    n_bins = S.shape[0]
    n_fft = 2 * (n_bins - 1)
    hop_length = n_fft // 4
    
    # Run Griffin-Lim
    print(f"Running Griffin-Lim with n_fft={n_fft}, hop_length={hop_length}")
    y = librosa.griffinlim(S, n_iter=64, hop_length=hop_length, win_length=n_fft)
    
    # Save the result
    sf.write('/workspace/reconstructed.wav', y, 22050)
    print("Saved to /workspace/reconstructed.wav")

if __name__ == "__main__":
    main()
