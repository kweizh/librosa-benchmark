import librosa
import numpy as np

sr = librosa.get_samplerate('/workspace/input.wav')
stream = librosa.stream('/workspace/input.wav', block_length=16, frame_length=2048, hop_length=512)

rms_list = []
durations = []
for i, block in enumerate(stream):
    rms = np.sqrt(np.mean(block**2))
    rms_list.append(rms)
    durations.append(len(block) / sr)

# Reconstruct
rms_list = np.array(rms_list)
durations = np.array(durations)
reconstructed_rms = np.sqrt(np.sum(rms_list**2 * durations) / np.sum(durations))

y, sr = librosa.load('/workspace/input.wav', sr=None)
global_rms = np.sqrt(np.mean(y**2))

print("Reconstructed:", reconstructed_rms)
print("Global:", global_rms)
print("Ratio:", reconstructed_rms / global_rms)

import librosa
import numpy as np
import csv

def main():
    input_path = '/workspace/input.wav'
    output_path = '/workspace/rms_stream.csv'
    
    block_length = 16
    frame_length = 2048
    hop_length = 512
    
    sr = librosa.get_samplerate(input_path)
    stream = librosa.stream(
        input_path,
        block_length=block_length,
        frame_length=frame_length,
        hop_length=hop_length
    )
    
    with open(output_path, 'w', newline='') as csvfile:
        writer = csv.writer(csvfile)
        writer.writerow(['block_index', 'start_time_sec', 'end_time_sec', 'rms'])
        
        for i, block in enumerate(stream):
            # Calculate RMS for the block
            rms = np.sqrt(np.mean(block**2))
            
            # Calculate start and end samples
            start_sample = i * block_length * hop_length
            end_sample = start_sample + len(block)
            
            # Convert to seconds
            start_time_sec = start_sample / sr
            end_time_sec = end_sample / sr
            
            writer.writerow([i, start_time_sec, end_time_sec, rms])

if __name__ == '__main__':
    main()
