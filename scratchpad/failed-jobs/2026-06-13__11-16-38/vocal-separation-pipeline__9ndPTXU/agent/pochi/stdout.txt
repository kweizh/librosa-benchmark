import librosa
import soundfile as sf
import numpy as np

# Load the input audio
y, sr = librosa.load('/workspace/input.wav', sr=None, mono=False)
print("Shape:", y.shape, "SR:", sr)

is_stereo = (y.ndim == 2)

def separate(y_mono):
    S_full, phase = librosa.magphase(librosa.stft(y_mono))
    
    # Ensure width is not larger than the number of frames
    width = min(int(librosa.time_to_frames(2, sr=sr)), S_full.shape[1] - 1)
    if width < 1:
        width = 1
        
    S_filter = librosa.decompose.nn_filter(S_full,
                                           aggregate=np.median,
                                           metric='cosine',
                                           width=width)
    
    S_filter = np.minimum(S_full, S_filter)
    
    mask = librosa.util.softmask(S_filter, S_full - S_filter, power=2)
    
    stft_complex = librosa.stft(y_mono)
    background_stft = mask * stft_complex
    foreground_stft = stft_complex - background_stft
    
    y_background = librosa.istft(background_stft, length=len(y_mono))
    y_foreground = librosa.istft(foreground_stft, length=len(y_mono))
    
    return y_foreground, y_background

if is_stereo:
    y_f_list = []
    y_b_list = []
    for ch in range(y.shape[0]):
        y_f, y_b = separate(y[ch])
        y_f_list.append(y_f)
        y_b_list.append(y_b)
    y_foreground = np.array(y_f_list)
    y_background = np.array(y_b_list)
else:
    y_foreground, y_background = separate(y)

sf.write('/workspace/foreground.wav', y_foreground.T if is_stereo else y_foreground, sr)
sf.write('/workspace/background.wav', y_background.T if is_stereo else y_background, sr)
