import librosa
import numpy as np

y_ref, sr = librosa.load('/workspace/reference.wav', sr=22050)
y_comp, _ = librosa.load('/workspace/comparison.wav', sr=22050)
chroma_ref = librosa.feature.chroma_cens(y=y_ref, sr=sr, hop_length=11025)
chroma_comp = librosa.feature.chroma_cens(y=y_comp, sr=sr, hop_length=11025)
_, wp = librosa.sequence.dtw(X=chroma_ref, Y=chroma_comp, metric='cosine')
wp_reversed = wp[::-1]

# Group comparison indices by reference index
ref_to_comp = {}
for r, c in wp_reversed:
    if r not in ref_to_comp:
        ref_to_comp[r] = []
    ref_to_comp[r].append(c)

# Let's try different mapping strategies
for strategy in ['mean', 'median', 'first', 'last']:
    comp_times = []
    for r in range(31):
        indices = ref_to_comp[r]
        if strategy == 'mean':
            val = np.mean(indices)
        elif strategy == 'median':
            val = np.median(indices)
        elif strategy == 'first':
            val = indices[0]
        elif strategy == 'last':
            val = indices[-1]
        
        # Convert frame to time
        comp_time = librosa.frames_to_time(val, sr=sr, hop_length=11025)
        comp_times.append(comp_time)
        
    monotonic = all(comp_times[i] <= comp_times[i+1] for i in range(len(comp_times)-1))
    print(f'{strategy} monotonic:', monotonic)
    print(f'{strategy} comp_times:', [round(t, 3) for t in comp_times])
