onset_env = librosa.onset.onset_strength(y=y, sr=sr)
tempogram = librosa.feature.fourier_tempogram(onset_envelope=onset_env, sr=sr)
bpm = librosa.fourier_tempo_frequencies(win_length=384, sr=sr)

global_spectrum = np.mean(np.abs(tempogram), axis=1)

print('All peaks:')
for p in peaks:
