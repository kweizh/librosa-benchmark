import librosa
import numpy as np
import scipy.signal

y, sr = librosa.load('/workspace/input.wav')
f0, vf, vp = librosa.pyin(y, fmin=librosa.note_to_hz('C2'), fmax=librosa.note_to_hz('C7'), sr=sr)
t = librosa.times_like(f0, sr=sr)

# Median filter on voiced F0 trace
voiced_f0 = f0[vf]
smoothed_voiced_f0 = scipy.signal.medfilt(voiced_f0, kernel_size=5)

# Place smoothed F0 back
f0_smoothed = np.full_like(f0, np.nan)
f0_smoothed[vf] = smoothed_voiced_f0

# Compute midi notes for each frame
midi_notes = np.full_like(f0, -1, dtype=int)
for i in range(len(f0)):
    if vf[i]:
        midi_notes[i] = int(round(12 * np.log2(f0_smoothed[i] / 440.0) + 69))

# Group consecutive voiced frames into notes
notes = []
current_note_frames = []

for i in range(len(f0)):
    if vf[i]:
        m = midi_notes[i]
        # Check if we need to cut current note
        if len(current_note_frames) > 0:
            prev_idx = current_note_frames[-1]
            prev_m = midi_notes[prev_idx]
            # Cut if unvoiced frame interrupted, or midi note changed
            # Since we iterate i sequentially, if prev_idx != i - 1, then there was an unvoiced frame in between
            if prev_idx != i - 1 or m != prev_m:
                # Save current note
                notes.append(current_note_frames)
                current_note_frames = [i]
            else:
                current_note_frames.append(i)
        else:
            current_note_frames.append(i)
    else:
        if len(current_note_frames) > 0:
            notes.append(current_note_frames)
            current_note_frames = []

if len(current_note_frames) > 0:
    notes.append(current_note_frames)

# Process notes
processed_notes = []
hop_duration = 512 / sr

for nf in notes:
    start_idx = nf[0]
    end_idx = nf[-1]
    
    start_time = t[start_idx]
    end_time = t[end_idx] + hop_duration
    
    # Mean F0
    note_f0s = f0_smoothed[nf]
    mean_f0 = float(np.mean(note_f0s))
    
    midi_note = int(midi_notes[start_idx])
    
    processed_notes.append({
        'start': float(start_time),
        'end': float(end_time),
        'midi_note': midi_note,
        'mean_f0_hz': mean_f0,
        'num_frames': len(nf)
    })

print(f'Total initial notes: {len(processed_notes)}')
for idx, pn in enumerate(processed_notes):
    print(f\"Note {idx}: start={pn['start']:.3f}, end={pn['end']:.3f}, dur={pn['end']-pn['start']:.3f}, midi={pn['midi_note']}, mean_f0={pn['mean_f0_hz']:.2f}, frames={pn['num_frames']}\")
