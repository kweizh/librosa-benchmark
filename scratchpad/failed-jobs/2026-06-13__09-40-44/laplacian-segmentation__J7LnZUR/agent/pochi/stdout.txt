import numpy as np
import scipy
import librosa

y, sr = librosa.load('/workspace/input.wav')
BINS_PER_OCTAVE = 12 * 3
N_OCTAVES = 7
C = librosa.amplitude_to_db(librosa.cqt(y=y, sr=sr,
                                        bins_per_octave=BINS_PER_OCTAVE,
                                        n_bins=N_OCTAVES * BINS_PER_OCTAVE),
                            ref=np.max)

tempo, beats = librosa.beat.beat_track(y=y, sr=sr, trim=False)
Csync = librosa.util.sync(C, beats, aggregate=np.median)

R = librosa.segment.recurrence_matrix(Csync, width=3, mode='affinity', sym=True)
df = librosa.segment.timelag_filter(scipy.ndimage.median_filter)
Rf = df(R, size=(1, 7))

mfcc = librosa.feature.mfcc(y=y, sr=sr)
Msync = librosa.util.sync(mfcc, beats)
path_distance = np.sum(np.diff(Msync, axis=1)**2, axis=0)
sigma = np.median(path_distance)
path_sim = np.exp(-path_distance / sigma)
R_path = np.diag(path_sim, k=1) + np.diag(path_sim, k=-1)

deg_path = np.sum(R_path, axis=1)
deg_rec = np.sum(Rf, axis=1)
mu = deg_path.dot(deg_path + deg_rec) / np.sum((deg_path + deg_rec)**2)
A = mu * Rf + (1 - mu) * R_path

L = scipy.sparse.csgraph.laplacian(A, normed=True)
evals, evecs = scipy.linalg.eigh(L)
evecs = scipy.ndimage.median_filter(evecs, size=(9, 1))
Cnorm = np.cumsum(evecs**2, axis=1)**0.5

print('Csync shape:', Csync.shape)

# Post-processing helper functions
def merge_short_segments(boundaries, min_duration=1.0):
    while True:
        changed = False
        for i in range(len(boundaries) - 1):
            dur = boundaries[i+1] - boundaries[i]
            if dur < min_duration:
                if i == len(boundaries) - 2:
                    if i > 0:
                        boundaries.pop(i)
                        changed = True
                        break
                else:
                    boundaries.pop(i+1)
                    changed = True
                    break
        if not changed:
            break
    return boundaries

def adjust_segment_count(boundaries, min_segments=3, max_segments=15, min_duration=1.0):
    boundaries = merge_short_segments(boundaries, min_duration)
    while len(boundaries) - 1 > max_segments:
        best_i = -1
        min_merged_dur = float('inf')
        for i in range(1, len(boundaries) - 1):
            merged_dur = boundaries[i+1] - boundaries[i-1]
            if merged_dur < min_merged_dur:
                min_merged_dur = merged_dur
                best_i = i
        if best_i != -1:
            boundaries.pop(best_i)
        else:
            break
            
    while len(boundaries) - 1 < min_segments:
        longest_i = -1
        max_dur = -1
        for i in range(len(boundaries) - 1):
            dur = boundaries[i+1] - boundaries[i]
            if dur > max_dur:
                max_dur = dur
                longest_i = i
        if longest_i != -1 and max_dur >= 2 * min_duration:
            midpoint = (boundaries[longest_i] + boundaries[longest_i+1]) / 2.0
            boundaries.insert(longest_i + 1, midpoint)
        else:
            break
    return boundaries
